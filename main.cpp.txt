#include <iostream>
#include "LabTasks.h"  // Підключення нашого заголовного файлу, де оголошені функції задач
#include <Windows.h>

using namespace std;

int main() { 
    // щоб в консолі адекватно відображалися та вводилися українські літери
    SetConsoleCP(1251);
    SetConsoleOutputCP(1251);

    int choice; // Змінна, де зберігатиметься номер пункту, який обере користувач

    // Цикл do-while гарантує, що меню покажеться хоча б один раз,
    // і буде крутитися по колу, поки користувач не введе '0' для виходу
    do {
        // Виведення текстового інтерфейсу головного меню
        cout << "\n========== ГОЛОВНЕ МЕНЮ ==========" << endl;
        cout << "1. Виконати завдання Param76 (Додати 1 секунду до 5 значень)" << endl;
        cout << "2. Виконати завдання Begin16 (Відстань між точками x1 та x2)" << endl;
        cout << "3. Виконати завдання Boolean17 (Перевірка на непарне тризначне)" << endl;
        cout << "0. Вихід з програми" << endl;
        cout << "==================================" << endl;
        cout << "Оберіть дію: ";

        cin >> choice; // Зчитування вибору користувача з консолі

        // Оператор switch аналізує значення змінної 'choice' і запускає відповідний блок коду
        switch (choice) {
        case 1:
            RunParam76(); // Якщо ввели 1 — викликаємо функцію обробки часу
            break;
        case 2:
            RunBegin16(); // Якщо ввели 2 — рахуємо відстань між точками
            break;
        case 3:
            RunBoolean17(); // Якщо ввели 3 — перевіряємо число на тризначність та непарність
            break;
        case 0:
            // Якщо ввели 0 — виводимо прощальне повідомлення і завершуємо цикл
            cout << "Роботу завершено успішно!" << endl;
            break;
        default:
            // Цей блок спрацює, якщо користувач ввів будь-яке інше число (наприклад, 5 чи 99)
            cout << "Помилка: такого пункту немає. Спробуйте ще раз." << endl;
        }
    } while (choice != 0); // Перевірка умови: якщо choice НЕ дорівнює 0, йдемо на нове коло

    return 0; // Повертаємо 0 — знак операційній системі, що програма завершилася без збоїв
}
